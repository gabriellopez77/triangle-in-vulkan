cmake_minimum_required(VERSION 3.30)

set(CMAKE_CXX_STANDARD 20) # c++ version
set(CMAKE_CXX_STANDARD_REQUIRED, TRUE) # required c++ version

# disable building GLFW examples, tests, and documentation
set(GLFW_BUILD_EXAMPLES OFF CACHE INTERNAL "")
set(GLFW_BUILD_TESTS OFF CACHE INTERNAL "")
set(GLFW_BUILD_DOCS OFF CACHE INTERNAL "")

project(LearnVulkan) # project name


# locate Vulkan
find_package(Vulkan REQUIRED)

# subdirectories
add_subdirectory("dependencies/glfw") # add glfw as a subdirectory
add_subdirectory("dependencies/glm") # add glm as a subdirectory


add_executable(${PROJECT_NAME}) # create executable with project name

file(GLOB_RECURSE MY_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}/src/*.cpp") # read all cpp files in src folder and save in MY_SOURCES variable

target_sources(${PROJECT_NAME} PRIVATE ${MY_SOURCES}) # add sources saved in MY_SOURCES variable to executable
# PRIVATE: only for this target, PUBLIC: for this and all targets that link to it, INTERFACE: only for targets that link to it

target_include_directories(${PROJECT_NAME} PRIVATE "${CMAKE_CURRENT_SOURCE_DIR}/src/*.h") # add include directories for header files
target_include_directories(${PROJECT_NAME} PRIVATE "${Vulkan_INCLUDE_DIRS}") # add include directories for Vulkan headers,
#Vulkan_INCLUDE_DIRS is a variable provided by find_package(Vulkan) that contains the include directories for Vulkan headers

target_link_libraries(${PROJECT_NAME} PRIVATE glfw) # link glfw library to executable
target_link_libraries(${PROJECT_NAME} PRIVATE Vulkan::Vulkan) # link Vulkan library to executable
target_link_libraries(${PROJECT_NAME} PRIVATE glm) # link glm library to executable

# macros
target_compile_definitions(${PROJECT_NAME} PUBLIC PROJECT_ROOT_PATH="${CMAKE_CURRENT_SOURCE_DIR}")
